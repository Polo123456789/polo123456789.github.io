Render

Ya aprendimos a poner imagenes en pantalla, y esta ves las hemos estado renderizando con software, realizando todos los calculos con el cpu. Ahora vamos a aprovechar la aceleracion por hardare que nos da la tarjeta grafica y pasarle la parte pesada a la misma. En parte tambien recomiendo usar esto ya que aun no he programado todas las funcionalidades con el rendering de software, solo las imagenes simples y sin nada mas. Una idea para tener en la parte de atras de la cabeza, igual que con las superficies y las ventanas, es que al trabajar con el renderer vamos a trabajar con texturas. Asi nos queda el codigo, y en realidad no hay demaciados cambios:

#define PS_UTIL_TIMER
#define PS_UTIL_LOG

#include <SDL.h> // En linux <SDL2/SDL.h>
#include "ps_utility.hpp"
#include "sDiamond.hpp"

#define SCR_W 640
#define SCR_H 480

int main(int argc, char* argv[]) {
	if (!sdmd::init()) {
		sdmd::log_error();
		sdmd::quit();
		return 1;
	}

	sdmd::Window ventana("Tutorial", SCR_W, SCR_H);				// R, G, B, A
	if (!ventana.createdSuccesfully() || !ventana.createRenderer(0x0, 0x0, 0x0, 0xff)) {
		sdmd::log_error();
		sdmd::quit();
		return 2;
	}

	sdmd::Image imagen("imagen.bmp", 64, 64);
	if (!imagen.createdSuccesfully() || !imagen.createTexture(ventana)) {
		sdmd::log_error();
		sdmd::quit();
		return 3;
	}

	sdmd::Event evento;
	bool corriendo = true;
	while(corriendo){
		while(sdmd::Event::poll(evento)) {
			switch(evento.type()) {
			case sdmd::QUIT:
				corriendo = false;
				break;
			default:
				break;
			}
		}

		ventana.clearRenderer();
		imagen.position.x = SCR_W/2 - imagen.position.w / 2;
		imagen.position.y = SCR_H/2 - imagen.position.h / 2;
		ventana.render(imagen);
		ventana.updateRenderer();
		sdmd::delay(33);
	}


	sdmd::quit();
	return 0;
}
